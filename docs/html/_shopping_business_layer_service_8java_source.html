272 /**
273 * Validates the login credentials for the specified user and password.
274 *
275 * @param user The username.
276 * @param pwd The password.
277 * @return true if the login credentials are valid, false otherwise.
278 */
279 public Boolean validateLogin(String user, String pwd) {
280 String inputPwdHash = CoreUtils.encodeSha256String(pwd);
281
282 UserDTO userDTO = getUser(user);
283 String userName = userDTO.getValue("email_address").toString();
284 if (userName == null) {
285 String msg = "Login attempt for user: " + user + " with password: " + inputPwdHash + " - Unknown User";
286 addAuditTrail(msg, userDTO);
287 return false;
288 }
289 String passwordHash = userDTO.getValue("password_hash").toString();
290
291 Boolean isValid = user.equals(userName) && inputPwdHash.equals(passwordHash);
292 String msg = "Login attempt for user: " + user + " with password: " + inputPwdHash + " - "
293 + (isValid ? "Success" : "Failure");
294 addAuditTrail(msg, userDTO);
295
296 return isValid;
297 }
